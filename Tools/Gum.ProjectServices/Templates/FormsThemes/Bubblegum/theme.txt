# Prerequisites for the Bubblegum Forms theme.
# Read by GumFormsPlugin.Services.ThemeRequirements at import time.
# Unknown keys are ignored (forward-compatible).

# Dynamic font generation. Bubblegum's fonts are sized at runtime and rely
# on KernSmith; BMFont can't produce equivalent output.
FontGenerator: KernSmith

# Skia-backed Standard elements. Required for the rounded chrome on
# Button/Window/etc. and the circular icons.
RequiredStandards: RoundedRectangle, ColoredCircle

# NuGet packages the user must add to their game project. The tool surfaces
# this list at the end of the import as a reminder; it does not edit the
# user's .csproj.
RuntimePackages: KernSmith.MonoGameGum (or .KniGum/.FnaGum), Gum.Themes.Bubblegum.MonoGame (or matching runtime)
